Python Sentiment Engine

At the heart of MindEase is a simple Python-based sentiment analyzer. It scans your message for emotional keywords and assigns a mood score — then passes that context to the AI for a truly empathetic reply.

Python 3 · Basic Level
# MindEase Sentiment Analyzer — Basic Python Implementation

def analyze_sentiment(text):
    """Detects mood from user message using keyword scoring."""

    # Keyword dictionaries with mood weights
    mood_keywords = {
        "stressed":  ["stress", "overwhelmed", "pressure", "exam", "deadline"],
        "anxious":   ["anxious", "worried", "nervous", "panic", "scared"],
        "sad":        ["sad", "lonely", "cry", "hopeless", "depressed"],
        "happy":      ["happy", "good", "great", "excited", "amazing"],
        "neutral":    ["okay", "fine", "alright", "normal"]
    }

    text_lower = text.lower()          # Step 1: lowercase
    scores = {mood: 0 for mood in mood_keywords}  # Step 2: init scores

    # Step 3: Count keyword matches
    for mood, keywords in mood_keywords.items():
        for word in keywords:
            if word in text_lower:
                scores[mood] += 1

    # Step 4: Return mood with highest score
    detected_mood = max(scores, key=lambda m: scores[m])
    confidence = scores[detected_mood] / (len(text_lower.split()) + 1)
    
    return {
        "mood": detected_mood,
        "score": round(confidence, 2),
        "all_scores": scores
    }

# Example usage
result = analyze_sentiment("I'm really stressed about my exam deadline")
print(result)
# → {'mood': 'stressed', 'score': 0.33, 'all_scores': {...}}
